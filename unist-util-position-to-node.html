export const atPos = pos => (node) => {
  const pos2 = node.position
  return (pos2.start.line >= pos.start.line
    && pos2.start.line <= pos.end.line)
    ||
    (pos2.end.line >= pos.start.line
     && pos2.end.line <= pos.end.line)
}
