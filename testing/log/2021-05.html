const toMarkdown = (x) => `- [](${x})`

const withList = async (list) => {
  return list
         .filter(startsWith)
         .map(addPrefix)
         .map(toMarkdown)
         .join('\n')
};

return lit.fs.readdir(prefix).then(withList);
