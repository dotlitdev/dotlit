const thisMonth = '2021-05-'
const prefix = '/testing/log/'

const startsWith = x => x.startsWith(thisMonth)
const addPrefix = x=>prefix+x

const withList = async list => {
   return list.filter(startsWith)
              .map(addPrefix)
}

return lit.fs
          .readdir(prefix)
          .then(withList)
