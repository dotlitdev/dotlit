const prefix = "/testing/log/";

const toMarkdown = ([x,y]) => `- [${x.split('.')[0]}](${y})`

const withList = async (list) => {
  return list
         .filter(startsWith)
         .map(x=>[x])
         .map(addPrefix)
         .map(toMarkdown)
         .join('\n')
};

return lit.fs.readdir(prefix).then(withList);
